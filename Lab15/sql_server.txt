//Tabla Materiales
create table Materiales(
Clave numeric(5),
Desccripcion varchar(50),
Costo numeric(8,2)
)


//Tabla Provedores
create table Provedores(
RFC varchar(12),
Razon_Social varchar(50),
)


//Tabla Proyectos
create table Proyectos(
Numero numeric(5),
Denominacion varchar(50),
)


//Tabla Entregan
create table Entregan(
Clave numeric(5),
RFC varchar(12),
Numero numeric(5),
Fecha datetime,
Cantidad numeric(8,2),
)
